keys:
    '1' : Monoatomic Mode 
    '2' : Diatomic Mode 
    '3' : Triatomic Mode 
    'a' : Add a particle
    'r' : Remove a prticle
    'k'/'l': draw / erase an orbit
    'd' : draw only an orbit
    'm'/'n' : change box size
    't' : increase box size by a 5 factor
    'j'/'h' : increase / decrease orbit sizes
    'e' : change algorythm -> Check this one!
    'i': Zoom in
    'o': Zoom out
    's' : change view	-> Nice!

    'F4' : Save to 'save.xml'
    'F5' : Load from 'save.xml'	-> Nice!

    '+'/'-' : change time step
    'Space' : pause
    'f' : Fullscreen

    'Q'/'esc' : Quit
